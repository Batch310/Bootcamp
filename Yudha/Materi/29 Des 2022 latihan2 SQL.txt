create table kota(
	id integer primary key,
	nama text,
	id_propinsi integer
)

create table propinsi(
	id integer primary key,
	nama text
)

--1 join / inner join
select 
	propinsi.nama as "Propinsi",
	kota.nama as "Kota"
from
	kota
inner join 
	propinsi
on
	kota.id_propinsi = propinsi.id;
	
--Alias nama tabel
select 
	p.nama as "Propinsi",
	k.nama as "Kota"
from
	kota k
inner join 
	propinsi p
on
	k.id_propinsi = p.id;

--Left join
select 
	p.nama as "Propinsi",
	k.nama as "Kota"
from
	kota k
left join 
	propinsi p
on
	k.id_propinsi = p.id;

--Right Join
select 
	p.nama as "Propinsi",
	k.nama as "Kota"
from
	kota k
right join 
	propinsi p
on
	k.id_propinsi = p.id;
	
-- full outer join / full join
select 
	p.nama as "Propinsi",
	k.nama as "Kota"
from
	kota k
full join 
	propinsi p
on
	k.id_propinsi = p.id;
